## 2. Код приложения

### 2.1. Модель (`models.py`)

```python
from django.db import models
from django.utils import timezone

class Note(models.Model):
    title = models.CharField(max_length=200, verbose_name="Заголовок")
    content = models.TextField(verbose_name="Содержание")
    created_at = models.DateTimeField(default=timezone.now)

    def __str__(self):
        return self.title
```

### 2.2. Представления (`views.py`)

```python
def note_list(request):
    notes = Note.objects.all()
    return render(request, 'notes/note_list.html', {'notes': notes})

def note_create(request):
    if request.method == 'POST':
        form = NoteForm(request.POST)
        if form.is_valid():
            form.save()
            return redirect('notes:note_list')
    else:
        form = NoteForm()
    return render(request, 'notes/note_create.html', {'form': form})

def note_edit(request, pk):
    note = get_object_or_404(Note, pk=pk)
    if request.method == 'POST':
        form = NoteForm(request.POST, instance=note)
        if form.is_valid():
            form.save()
            return redirect('notes:note_detail', pk=note.id)
    else:
        form = NoteForm(instance=note)
    return render(request, 'notes/note_edit.html', {'form': form, 'note': note})
```

### 2.3. URL-маршруты (`urls.py`)

```python
urlpatterns = [
    path('', views.note_list, name='note_list'),
    path('create/', views.note_create, name='note_create'),
    path('detail/<int:pk>/', views.note_detail, name='note_detail'),
    path('edit/<int:pk>/', views.note_edit, name='note_edit'),
    path('delete/<int:pk>/', views.note_delete, name='note_delete'),
]
```

### 2.4. Переключатель темы (JavaScript)

```javascript
const toggle = document.getElementById('themeToggle');
if (localStorage.getItem('theme') === 'dark') {
    document.body.classList.add('dark');
    toggle.checked = true;
}
toggle.addEventListener('change', function() {
    if (this.checked) {
        document.body.classList.add('dark');
        localStorage.setItem('theme', 'dark');
    } else {
        document.body.classList.remove('dark');
        localStorage.setItem('theme', 'light');
    }
});
```
